{\color{solution}
\begin{enumerate}
	\item First we note that as 
	\begin{align*}
	|x| = \begin{cases}
	x: x \geq 0\\
	-x : x < 0
	\end{cases}
	\end{align*}
	$x \leq |x|$.\\
	\textbf{Case 1} ($x = -y$): $\sqrt{ }$\\	
	\textbf{Case 2} ($x > -y$):
	\begin{align*}
	|x+y| = x+y \leq |x| + |y|.
	\end{align*}
	\textbf{Case 3} ($x < -y \Leftrightarrow -x >- (-y)$):
	\begin{align*}
	|x+y| = |-x-y| = -x-y \leq |-x| + |-y| = |x| + |y|.
	\end{align*}
	
	\item 
	\begin{align*}
	|x| = |x - y + y| \leq |x-y| + |y|\\
	\Leftrightarrow |x| - |y| \leq |x-y|
	\end{align*}
	By interchanging $x$ and $y$ we hence obtain the assertion.
\end{enumerate}
