Vorlesung

\begin{align*}
&\frac{|z - \overline{z}|}{|z|} \leq \left( \frac{1}{2}\right)^N < \varepsilon \\
& \Leftrightarrow N < \frac{\log(\varepsilon)}{\log(\frac{1}{2})}
\end{align*}

\begin{verbatim}
>>> eps = 1
>>> while 1+eps > 1:
...     eps /= 2
>>> eps
1.1102230246251565e-16
\end{verbatim}
